## `src/config.ts`

<RunnableCodeBlock blockId="config-snippet">

```ts filename="src/config.ts"
export function loadConfig() {
  const apiKey = process.env.OPENAI_API_KEY?.trim() ?? '';

  return {
    baseUrl: process.env.OPENAI_BASE_URL?.trim() || 'https://api.openai.com/v1',
    apiKey,
    model: process.env.OPENAI_MODEL?.trim() || 'gpt-4.1-mini',
    useMock: apiKey.length === 0
  };
}
```

</RunnableCodeBlock>

## `src/main.ts`

<RunnableCodeBlock blockId="main-ts-snippet">

```ts filename="src/main.ts"
async function main() {
  const prompt = '请介绍一下 Agent 是什么';
  const config = loadConfig();
  const client = new OpenAiChatClient(config);
  const reply = await client.chat(prompt);

  console.log(`User: ${prompt}`);
  console.log(`Assistant: ${reply}`);
}
```

</RunnableCodeBlock>
